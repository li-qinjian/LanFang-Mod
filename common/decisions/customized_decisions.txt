lan_move_capital_to_malaya = {
	is_shown = {
		c:LAN ?= ROOT
		#NOT = { has_variable = moved_capital_to_malaya }
		NOT = {
			capital = { state_region = s:STATE_MALAYA }
		}
	}

	possible = {
		c:LAN ?= ROOT
		owns_entire_state_region = STATE_MALAYA
	}

	when_taken = {
		#set_variable = moved_capital_to_malaya
		set_capital = STATE_MALAYA
		set_state_type = incorporated
	}

	ai_chance = {
		value = 0
	}
}

lan_move_chinese_to_capital = {
	is_shown = {
		c:LAN ?= ROOT

			is_incorporated = no
			is_split_state = yes
			any_scope_pop = {
				culture = {
					has_discrimination_trait = heritage_han
				}
			}
		}
	}

	possible = {
		c:LAN ?= ROOT
		any_scope_state = {
			is_incorporated = no
			is_split_state = yes
			any_scope_pop = {
				culture = {
					has_discrimination_trait = heritage_han
				}
			}
		}
	}

	when_taken = {
		every_scope_state = {

			}

					}

					text = move_chineses_tt
					move_pop = root.capital
				}
			}
		}
	}

	ai_chance = {
		value = 0
	}
}

ban_opium = {
	is_shown = {
		is_player = yes
		exists = market
		market.owner = root
		NOT = { is_banning_goods = g:opium }
	}

	possible = {
		is_player = yes
		NOT = { has_law_or_variant = law_type:law_free_trade }
		exists = market
		any_primary_culture = {
			has_discrimination_trait = heritage_han
		}
	}

	when_taken = {
		#add_banned_goods = g:opium
		if = {
			limit = {
				exists = c:DEI
				exists = c:SMB
				c:SMB = {
					is_subject_of = c:DEI
				}
			}
			c:DEI = {
				create_diplomatic_play = {
					name = dp_annex_subject
					target_country = c:SMB
					war = no
					type = dp_annex_subject
				}
			}
		}
	}

    ai_chance = {
		value = 0
    }
}

trigger_prussia_danish_war = {
	is_shown = {
		is_player = yes
		#game_date >= 1848.1.1
		NOT = { has_global_variable = prussia_danish_war_triggered }
		
		s:STATE_SCHLESWIG_HOLSTEIN = {
			any_scope_state = {
				OR = {
					owner = c:DEN
					owner = {
						is_subject_of = c:DEN
					}
				}
			}
		}
	}

	possible = {
		exists = c:PRU
		exists = c:DEN
		c:PRU = {
			is_at_war = no
			NOT = {
				has_diplomatic_pact = {
					who = c:DEN
					type = alliance
				}
			}
		}
		c:DEN = {
			is_at_war = no
		}
	}

	when_taken = {
		set_global_variable = prussia_danish_war_triggered
		
		c:PRU = {
			create_diplomatic_play = {
				name = dp_liberate_subject
				target_country = c:HOL
				war = no
				type = dp_liberate_subject
				
				add_war_goal = {
					holder = c:PRU
					type = liberate_subject
					target_country = c:SCH
				}
			}
		}
		
		custom_tooltip = {
			text = prussia_danish_war_triggered_tt
		}
	}

	ai_chance = {
		value = 0
	}
}
